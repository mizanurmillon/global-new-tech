<?php
namespace App\Http\Controllers\Web\Backend\Projects;

use App\Http\Controllers\Controller;
use App\Models\Project;
use App\Models\ServiceArea;
use Illuminate\Http\Request;
use Yajra\DataTables\DataTables;

class ProjectController extends Controller
{
    // Project List / DataTable

    public function index(Request $request)
    {
        if ($request->ajax()) {
            $query = Project::with('serviceAreas')->latest()->get();

            return DataTables::of($query)
                ->addIndexColumn()
                ->addColumn('title', fn($project) => $project->title)
                ->addColumn('description', fn($project) => $project->description)
                ->addColumn('image', function ($project) {
                    return $project->image
                        ? '<img src="' . asset($project->image) . '" style="width:50px;height:50px;object-fit:cover;">'
                        : '-';
                })
                ->addColumn('action', function (Project $data) {
                    return view('components.action-buttons', [
                        'id'     => $data->id,
                        'show'   => 'admin.projects.show',
                        'edit'   => 'admin.projects.edit',
                        'delete' => true,
                    ])->render();
                })
                ->rawColumns(['image', 'action'])
                ->make(true);
        }

        return view('backend.layouts.projects.index');
    }

    // Create Project Form
    public function create()
    {
        $serviceAreas = ServiceArea::all();
        return view('backend.layouts.projects.create', compact('serviceAreas'));
    }

    // Store New Project
    public function store(Request $request)
    {
        $validated = $request->validate([
            'title'           => 'required|string|max:255',
            'description'     => 'nullable|string',
            'image'           => 'nullable|image|mimes:jpeg,png,jpg,gif,svg|max:20480',
            'services_area'   => 'required|array|min:1',
            'services_area.*' => 'exists:service_area,id',
        ]);

        $serviceAreas = $validated['services_area'];
        unset($validated['services_area']);

        // Upload image
        if ($request->hasFile('image')) {
            $validated['image'] = uploadFile($request->file('image'), 'uploads/projects');
        }

        $project = Project::create($validated);

        // Attach services
        $project->serviceAreas()->sync($serviceAreas);

        return redirect()->route('admin.projects.index')->with('success', 'Project created successfully.');
    }

    // Show Project
    public function show(Project $project)
    {
        $project->load('serviceAreas');
        return view('backend.layouts.projects.show', compact('project'));
    }

    // Edit Project Form
    public function edit(Project $project)
    {
        $project->load('serviceAreas');
        $serviceAreas = ServiceArea::all();

        return view('backend.layouts.projects.edit', compact('project', 'serviceAreas'));
    }

    // Update Project
    public function update(Request $request, Project $project)
    {
        // Validation
        $request->validate([
            'title'           => 'required|string|max:255',
            'description'     => 'nullable|string',
            'services_area'   => 'nullable|array',
            'services_area.*' => 'exists:service_area,id',
            'image'           => 'nullable|image|mimes:jpeg,png,jpg,gif,webp|max:2048',
        ]);

        // Update basic fields
        $project->title       = $request->title;
        $project->description = $request->description;

        // Update image if uploaded
        if ($request->hasFile('image')) {
            deleteFile($project->image);
            $project->image = uploadFile($request->file('image'), 'uploads/projects');
        }

        $project->save();

        // Sync services areas
        if ($request->has('services_area')) {
            $project->serviceAreas()->sync($request->services_area);
        } else {
            $project->serviceAreas()->sync([]);
        }

        return redirect()
            ->route('admin.projects.index')
            ->with('success', 'Project updated successfully!');
    }

    // Delete Project
    public function destroy(Project $project)
    {
        if ($project->image) {
            deleteFile($project->image);
        }

        $project->serviceAreas()->sync([]);
        $project->delete();

        return response()->json(['message' => 'Project deleted successfully.']);
    }
}

<?php
namespace App\Http\Requests\Industry;

use Illuminate\Foundation\Http\FormRequest;
use Illuminate\Support\Str;

class IndustryRequest extends FormRequest
{
    public function authorize(): bool
    {
        return true;
    }

    public function rules(): array
    {
        $id = $this->route('industry')?->id;

        return [
            'industrie_list_id'   => 'required|exists:industrie_lists,id|unique:industries,industrie_list_id,' . $id,

            // MAIN SECTION
            'main_title'               => 'required|string|max:255',
            'title'               => 'required|string|max:255',
            'subtitle'            => 'required|string|max:255',
            'description'         => 'required|string',
            'slug'                => 'required|string|max:255|unique:industries,slug,' . $id,
            'image'               => 'nullable|image',

            // HERO SECTION
            'hero_title'          => 'required|string|max:255',
            'hero_description'    => 'required|string',
            'hero_image'          => 'nullable|image',
            'hero_button_text'  => 'nullable|string|max:255',
            'hero_button_link'  => 'nullable|string',

            // GALLERY
            'gallery_title'       => 'required|string|max:255',
            'gallery_description' => 'required|string',
            'gallery_images.*'    => 'image',

            // SUB FOOTER
            'footer_image'        => 'nullable|image',
            'footer_main_title'   => 'nullable|string|max:255',
            'footer_title'        => 'required|string|max:255',
            'footer_description'  => 'required|string',
            'footer_button_text'  => 'nullable|string|max:255',
            'footer_button_link'  => 'nullable|string',

            'is_active'           => 'required|boolean',
        ];
    }

    public function messages(): array
    {
        return [
            'industrie_list_id.unique'=> 'The selected industry Category is already associated with another industry.',
        ];
    }

    protected function prepareForValidation()
    {
        $this->merge([
            'slug'      => Str::slug(trim(strtolower($this->slug))),
            'is_active' => $this->has('is_active'),
        ]);
    }
}
